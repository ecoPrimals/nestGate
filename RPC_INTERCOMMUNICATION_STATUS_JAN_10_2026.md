**Server Implementation**:
```rust
#[derive(Clone)]
pub struct NestGateRpcService {
    storage: Arc<dyn UniversalStorageProvider>,
    registry: Arc<ServiceRegistry>,
}

#[tarpc::server]
impl NestGateRpc for NestGateRpcService {
    async fn create_dataset(
        self,
        _context: tarpc::context::Context,
        name: String,
        params: DatasetParams,
    ) -> Result<DatasetInfo, NestGateError> {
        // Real implementation using storage backend
        self.storage.create_dataset(&name, params).await
    }
    
    // ... other real implementations
}

pub async fn serve_tarpc(
    addr: SocketAddr,
    service: NestGateRpcService,
) -> Result<()> {
    // Real server setup
    let listener = tcp::listen(addr, Json::default).await?;
    listener
        .filter_map(|conn| async move { conn.ok() })
        .map(BaseChannel::with_defaults)
        .map(|channel| {
            let server = NestGateRpcServer::new(server::Config::default(), service.clone());
            channel.execute(server.serve())
        })
        .buffer_unordered(100)
        .for_each(|_| async {})
        .await;
    Ok(())
}
```
